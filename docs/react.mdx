### 1. Install 
from root of your React app run the following on your command line 
<CodeGroup>
```shell npm 
npm install @swishjam/react
```
```shell yarn 
yarn add @swishjam/react
```
```shell pnpm 
pnpm add @swishjam/react
```
</CodeGroup>

### 2. Import into your App 
in your _app.js or top level layout import and setup Swishjam with your API key 
```javascript _app.js 
import { SwishjamProvider } from '@swishjam/react';

export default function App({ Component, pageProps }) {

    return (
        <SwishjamProvider apiKey='swishjam-XXXXX'>
            <Component {...pageProps} />
        </SwishjamProvider>
    )
} 
```

Get your api key from the <a target="_blank" href="https://app.swishjam.com/settings">Swishjam settings</a>. We have multiple API keys to separate data from your various sites & apps. We automatically create 2 keys for "marketing" and "product" for you to use.

### 3. Triggering Events 
Page views happen automatically so there's no need to trigger those, but if you'd like track specific events like a user successfully completing an action you can do the following

- import Swishjam into your component
- trigger your event 

<CodeGroup>
```javascript Simple Swishjam Import 
import { swishjam } from '@swishjam/react';

export default function YourComponent({}) {

    const clickHandler = (e) => {
        swishjam.event('event name', {
            ...include your data here
        })
    }

    return (
        <button onClick={(e) => clickHandler(e)}>
            Click This Button
        </button>        
    )
}
```
```javascript React Hooks useSwishjam 
import { useSwishjam } from '@swishjam/react';

export default function YourComponent({}) {
    const swishjam = useSwishjam();

    const clickHandler = (e) => {
        swishjam.event('event name', {
            ...include your data here
        })
    }

    return (
        <button onClick={(e) => clickHandler(e)}>
            Click This Button
        </button>        
    )
}
```
</CodeGroup>

### 4. Identifying Users
  
If your users login and you would like to identify their activity in Swishjam you can use the .indentify function 

- import Swishjam into your component
- identify your user 

<CodeGroup>
```javascript Standard Swishjam Import 
import { swishjam } from '@swishjam/core';

export default function YourComponent({}) {

    const clickHandler = (e) => {
        swishjam.identify('user@example.com', {
            ...include your data here
        })
    }

    return (
        <button onClick={(e) => clickHandler(e)}>
            Login
        </button>        
    )
}
```
```javascript React Hooks useSwishjam 
import { useSwishjam } from '@swishjam/react';

export default function YourComponent({}) {
    const swishjam = useSwishjam();

    const clickHandler = (e) => {
        swishjam.event('event name', {
            ...include your data here
        })
    }

    return (
        <button onClick={(e) => clickHandler(e)}>
            Click This Button
        </button>        
    )
}
```
</CodeGroup>

### 5. Grouping Users 

In many SaaS tools users are part of a larger grouping like a business or entity. Use groupings to automatically aggregate user data to a group

- import Swishjam into your component
- group your user

<CodeGroup>
```javascript Standard Swishjam Import
import { swishjam } from '@swishjam/core';

export default function YourComponent({}) {

    const clickHandler = (e) => {

            Login
```

### 6. Logging out a users

When your users log out you can trigger our log out which will help reset the device identifier. We'll disassociate all future events from the currently identified user

- import Swishjam into your component
- logout your user

<CodeGroup>
```javascript Standard Swishjam Import 
import { swishjam } from '@swishjam/core';

export default function YourComponent({}) {

    const clickHandler = (e) => {
        swishjam.logout()
    }

    return (
        <button onClick={(e) => clickHandler(e)}>
            Logout
        </button>        
    )
}
```
```javascript React Hooks useSwishjam
import { useSwishjam } from '@swishjam/react';

export default function YourComponent({}) {
    const swishjam = useSwishjam();

    const clickHandler = (e) => {
        swishjam.logout()
    }

    return (
        <button onClick={(e) => clickHandler(e)}>
            Click This Button
        </button>
    )
}
```
</CodeGroup>
